import random

def play_hangman():
    # Predefined list of words
    words = ["python", "code", "alpha", "logic", "script"]
    word_to_guess = random.choice(words)
    guessed_letters = []
    attempts_left = 6

    print("--- Welcome to Hangman! ---")
    print("Guess the secret word, one letter at a time.")
    print(f"You have {attempts_left} incorrect attempts allowed.\n")

    while attempts_left > 0:
        # Display current state of the word (e.g., "p y _ _ o n")
        display_word = [letter if letter in guessed_letters else "_" for letter in word_to_guess]
        print("Word: " + " ".join(display_word))
        
        # Check if user has guessed all letters
        if "_" not in display_word:
            print(f"\n🎉 Congratulations! You guessed the word: '{word_to_guess}'")
            return

        guess = input("Enter a letter: ").lower().strip()

        # Input validation
        if len(guess) != 1 or not guess.isalpha():
            print("⚠️ Please enter a single valid letter.\n")
            continue

        if guess in guessed_letters:
            print("⚠️ You already guessed that letter. Try another one!\n")
            continue

        guessed_letters.append(guess)

        if guess in word_to_guess:
            print(f"✅ Good job! '{guess}' is in the word.\n")
        else:
            attempts_left -= 1
            print(f"❌ Wrong guess! '{guess}' is not in the word. Remaining attempts: {attempts_left}\n")

    print(f"💥 Game Over! You ran out of attempts. The word was: '{word_to_guess}'")

if __name__ == "__main__":
    play_hangman()
